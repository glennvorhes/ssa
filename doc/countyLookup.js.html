/**
 * 
 * @param {string|number} countyId
 * @returns {string}
 */
export function getCountyById(countyId){
    "use strict";
    
    if (typeof countyId == 'number'){
        countyId = countyId.toFixed();
    }
    
    return countyLookup[countyId];
}

nm.getCountyById = getCountyById;
